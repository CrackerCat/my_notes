- 相互转换摄氏与华氏温度，生成 n 阶斐波那契数列。

```rust
fn main() {
    let hot = 212.0;
    let cold = 0.0;
    println!("{}°F = {}°C", hot, fahrenheit_to_celsius(hot));
    println!("{}°C = {}°F", cold, celsius_to_fahrenheit(cold));
    fibonacci(6);
}

// 转换摄氏温度到华氏温度
fn celsius_to_fahrenheit(celsius: f64) -> f64 {
    celsius * 1.8 + 32.0
}

// 从华氏温度转换到摄氏温度
fn fahrenheit_to_celsius(fahrenheit: f64) -> f64 {
    (fahrenheit - 32.0) / 1.8
}

// 打印n阶斐波那契数列
fn fibonacci(n: i32) {
    let mut a = 0;
    let mut b = 1;
    let mut i = 0;
    while i < n {
        print!("{} ", b);
        let c = a + b;
        a = b;
        b = c;
        i += 1;
    }
    println!();
}
```
